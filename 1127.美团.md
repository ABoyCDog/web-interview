- 单例模式

  这种设计模式思想是确保一个类只有唯一实例，一般用于全局缓存，比如全局window、唯一登录浮窗等。采用闭包的方式实现如下：
  ```js
  var single = (function() {
    let instance

    function getInstance() {
      // 如果该实例存在，则直接返回，否则就对其实例化
      if(instance === undefined) {
        instance = new Construct()
      }
      return instance
    }

    function Constuct() {
      // 生成单例模式
    }

    return {
      getInstance : getInstance
    }
  })
  ```
